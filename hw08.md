## 2）C语言实现（先用注释写好算法，然后翻译）  
#include<stdio.h>  
void A (int num) {  
  int a;  
  a = num % 16;  
  if (num >= 16)  
    A(num / 16);  
  switch(a) {  
    case 10:  
        printf("A");  
        break;  
    case 11:  
        printf("B");  
        break;
    case 12:
        printf("C");
        break;
    case 13:
        printf("D");
        break;
    case 14:
        printf("E");
        break;
    case 15:
        printf("F");
        break;
    default:
        printf("%d", a);
  } 
}
 int main(void) {
  int num;
  scanf("%d", &num);
  A(num);
  return 0;
}
